  • Fetch封装——reco-fetch

    • 安装
      • 配置
        • 案例

        TIP

        • fetch 必然要替换 XMLHttpRequest ，所以是时候尝试 fetch 了；
        • 本封装仅针对npm引入；
        • 本封装依赖 es6-promise 和 whatwg-fetch，分别对 promise 和 fetch 进行兼容性处理；
        • 还有一种兼容性处理是依赖 es6-promise 和isomorphic-fetch ，但是看它的源码就会发现，isomorphic-fetch 只不过是引用了 whatwg-fetch ，并没有做二次开发，isomorphic-fetch 只是将fetch添加为全局，以便其API在客户端和服务器之间保持一致，所以没必要。

        适用于客户端的fetch封装，使用了ES6语法，或许你需要babel进行转译。

        # 安装

        $ npm isntall reco-fetch

        # 配置

        除了下面给出的参数，其他参数请结合 MDN

        import recoFetch from 'reco-fetch'

/**
 * @param {String, must} url         接口URL
 * @param {Object, must} options     参数对象，包括:
 *        method  {String, optional} 请求方式，默认'get'
 *        headers {Object, optional} 设置请求头
 *        params  {Object, optional} URL参数对象
 *        body    {Object, optional} 请求体参数对象
 *        timeout {Number, optional} 请求超时
 *        type    {String, optional} 'post'请求时，可以设置：'json'、'formData'
 */

const options = {
  method: 'post',
  headers: {},
  timeout: 1000,
  body: {
    id: 1,
    value: 2
  }
}

recoFetch(url, options). then(res => {
  console.log(res)
}).catch(err => {
  console.log(err)
})

        GET

  method: 'get',

    value: 2
  }
}

recoFetch(url, options). then(res => {
  console.log(res)
}).catch(err => {
  console.log(err)
})

        POST JSON

        const options = {
  method: 'post',
  body: {
    key: 1,
    value: 2
  },
  type: 'json'
}

recoFetch(url, options). then(res => {
  console.log(res)

  console.log(err)

        POST formData

        const options = {
  method: 'post',
  body: {
    key: 1,
    value: 2
  },
  type: 'formData'
}

// 或

const form = document.querySelector('form')
const options = {
  method: 'post',
  body: form
}

recoFetch(url, options). then(res => {
  console.log(res)
}).catch(err => {
  console.log(err)
})

        uploadFile

        const fileField = document.querySelector("input[type='file']")

const options = {
  method: 'post',
  body: {
    file: fileField.files[0]
  },
  type: 'formData'
}

// 或

const formData = new FormData()
const fileField = document.querySelector("input[type='file']")

formData.append('file', fileField.files[0])

const options = {
  method: 'post',
  body: formData
}

recoFetch(url, options). then(res => {
  console.log(res)
}).catch(err => {
  console.log(err)
})
